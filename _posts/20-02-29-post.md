```cpp
    cin >> n;

    for(int i = 0; i < n; i ++){
        cin >> temp;
        arr[i] = temp;
        arr2[i] = 0;
    }

    int ans = INT_MAX;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            arr2[i] = arr2[i] + arr[j]*2*(i+j+absolValue(i-j));

        ans = min(ans, arr2[i]);

    cout << ans;
```
